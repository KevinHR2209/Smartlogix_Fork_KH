CREATE TABLE "pedido" (
  "id_pedido" bigserial PRIMARY KEY,
  "id_cliente" bigint, /* REFERENCIA LÓGICA al MS Clientes */
  "fecha_creacion" timestamptz DEFAULT now(),
  "monto_total" int,
  "estado_pedido" varchar(50)
);

CREATE TABLE "detalle_pedido" (
  "id_detalle" bigserial PRIMARY KEY,
  "id_pedido" bigint,
  "id_presentacion" bigint, /* REFERENCIA LÓGICA al MS Inventario */
  "cantidad" int,
  "precio_unitario_snapshot" int
);

CREATE TABLE "pago" (
  "id_pago" bigserial PRIMARY KEY,
  "id_pedido" bigint,
  "monto_transaccion" int,
  "metodo_pago" varchar(50),
  "estado_pago" varchar(50),
  "fecha_pago" timestamptz,
  "token_transaccion" varchar(255)
);

-- Claves foráneas internas del MS Ventas
ALTER TABLE "detalle_pedido" ADD FOREIGN KEY ("id_pedido") REFERENCES "pedido" ("id_pedido") DEFERRABLE INITIALLY IMMEDIATE;
ALTER TABLE "pago" ADD FOREIGN KEY ("id_pedido") REFERENCES "pedido" ("id_pedido") DEFERRABLE INITIALLY IMMEDIATE;
